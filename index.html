IoT Smart Lighting System

Controls lighting via mobile app and sensors for energy efficiency and convenience.

Python 
import asyncio
import json
from flask import Flask, jsonify
from flask_socketio import SocketIO, emit
import RPi.GPIO as GPIO

app = Flask(__name__)
socketio = SocketIO(app)

# GPIO setup for LED control
GPIO.setmode(GPIO.BCM)
LED_PINS = [18, 19, 20, 21]
for pin in LED_PINS:
    GPIO.setup(pin, GPIO.OUT)

@socketio.on('light_control')
def handle_light_control(data):
    led_id = data['led_id']
    state = data['state']
    GPIO.output(LED_PINS[led_id], GPIO.HIGH if state else GPIO.LOW)
    emit('status', {'led_id': led_id, 'state': state})

if __name__ == '__main__':
    socketio.run(app, host='0.0.0.0', port=5000)

Python Network Monitoring Tool

Visualizes bandwidth in real time for network performance analysis.

Python 
import psutil
import time
import matplotlib.pyplot as plt
from datetime import datetime
import json

class NetworkMonitor:
    def __init__(self):
        self.data_history = []
        
    def get_network_stats(self):
        # Get network I/O statistics
        net_io = psutil.net_io_counters()
        bytes_sent = net_io.bytes_sent
        bytes_recv = net_io.bytes_recv
        
        # Calculate bandwidth (bytes per second)
        upload_speed = bytes_sent / (1024 * 1024)  # MB/s
        download_speed = bytes_recv / (1024 * 1024)  # MB/s
        
        return {
            'timestamp': datetime.now().isoformat(),
            'upload_speed': upload_speed,
            'download_speed': download_speed,
            'bytes_sent': bytes_sent,
            'bytes_recv': bytes_recv
        }
    
    def visualize_data(self):
        timestamps = [d['timestamp'] for d in self.data_history]
        upload_speeds = [d['upload_speed'] for d in self.data_history]
        download_speeds = [d['download_speed'] for d in self.data_history]
        
        plt.figure(figsize=(12, 6))
        plt.plot(timestamps, upload_speeds, label='Upload Speed')
        plt.plot(timestamps, download_speeds, label='Download Speed')
        plt.xlabel('Time')
        plt.ylabel('Speed (MB/s)')
        plt.title('Network Performance Monitor')
        plt.legend()
        plt.xticks(rotation=45)
        plt.tight_layout()
        plt.show()

monitor = NetworkMonitor()
while True:
    stats = monitor.get_network_stats()
    monitor.data_history.append(stats)
    if len(monitor.data_history) > 100:
        monitor.visualize_data()
        break
    time.sleep(1)

Digital Marketing Dashboard

React + Firebase project for tracking marketing campaigns and analytics.

JavaScript 
import React, { useState, useEffect } from 'react';
import { collection, query, where, orderBy, onSnapshot } from 'firebase/firestore';
import { Line, Bar, Pie } from 'recharts';

const MarketingDashboard = () => {
  const [campaigns, setCampaigns] = useState([]);
  const [analytics, setAnalytics] = useState({
    totalClicks: 0,
    conversions: 0,
    revenue: 0
  });

  useEffect(() => {
    // Fetch campaigns from Firebase
    const q = query(
      collection('campaigns'),
      orderBy('createdAt', 'desc'),
      limit(10)
    );
    
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const campaignData = snapshot.docs.map(doc => ({
        id: doc.id,
        ...doc.data()
      }));
      setCampaigns(campaignData);
    });

    return () => unsubscribe();
  }, []);

  const calculateROI = (cost, revenue) => {
    return ((revenue - cost) / cost * 100).toFixed(2);
  };

  return (
    <div className="dashboard">
      <h1>Marketing Analytics</h1>
      <div className="metrics-grid">
        <MetricCard 
          title="Total Campaigns" 
          value={campaigns.length} 
        />
        <MetricCard 
          title="Total Revenue" 
          value={`$${analytics.revenue.toLocaleString()}`} 
        />
        <MetricCard 
          title="Conversion Rate" 
          value={`${calculateROI(analytics.cost, analytics.revenue)}%`} 
        />
      </div>
      <CampaignChart data={campaigns} />
    </div>
  );
};

export default MarketingDashboard;

# Video Processing Pipeline

import os
import subprocess
from pathlib import Path

class VideoProcessor:
    def __init__(self, input_folder, output_folder):
        self.input_folder = Path(input_folder)
        self.output_folder = Path(output_folder)
        self.output_folder.mkdir(exist_ok=True)
        
    def process_videos(self):
        for video_file in self.input_folder.glob("*.mp4"):
            print(f"Processing: {video_file.name}")
            
            # Apply color grading
            color_preset = "cinematic_look"
            self.apply_color_grading(video_file, color_preset)
            
            # Add motion graphics
            self.add_motion_graphics(video_file)
            
            # Export in multiple formats
            self.export_formats(video_file)
    
    def apply_color_grading(self, video_file, preset):
        cmd = [
            "ffmpeg",
            "-i", str(video_file),
            "-vf", f"colorgrade=preset={preset}",
            "-c:v", "libx264",
            "-crf", "18",
            f"{self.output_folder}/{video_file.stem}_graded.mp4"
        ]
        subprocess.run(cmd)
    
    def add_motion_graphics(self, video_file):
        # Add lower thirds, transitions, and effects
        graphics_template = "corporate_branding"
        cmd = [
            "ffmpeg",
            "-i", str(video_file),
            "-vf", f"movie=overlay.png:format=rgba,loop=1,setpts=PTS-START+{len(video_file)*30/TB-1[v];[0][v]overlay[v]",
            "-c:v", "libx264",
            f"{self.output_folder}/{video_file.stem}_with_graphics.mp4"
        ]
        subprocess.run(cmd)
    
    def export_formats(self, video_file):
        formats = [
            ("mp4", "h264"),
            ("webm", "libvpx-vp9"),
            ("mov", "prores_ks")
        ]
        
        for format, codec in formats:
            output_file = self.output_folder / f"{video_file.stem}.{format}"
            cmd = [
                "ffmpeg",
                "-i", str(video_file),
                "-c:v", codec,
                "-crf", "18",
                str(output_file)
            ]
            subprocess.run(cmd)

# Usage
processor = VideoProcessor("raw_footage/", "final_videos/")
processor.process_videos()
